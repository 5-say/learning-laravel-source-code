Illuminate/Foundation/Application.php

/**
 * Create a new Illuminate application instance.
 * 创建一个新的 Illuminate 应用程序实例。
 *
 * @param  string|null  $basePath 应用程序基础路径（根目录）
 * @return void
 */
public function __construct($basePath = null)
{
    $this->registerBaseBindings(); // 注册基础绑定

    $this->registerBaseServiceProviders(); // 注册所有的基础服务提供商

    $this->registerCoreContainerAliases(); // 注册核心类别名

    if ($basePath) $this->setBasePath($basePath); // 设置基础路径
}

/**
 * Register the basic bindings into the container.
 * 在容器中注册基础绑定
 *
 * @return void
 */
protected function registerBaseBindings()
{
    static::setInstance($this); // 设置共享的容器实例

    $this->instance('app', $this);

    $this->instance('Illuminate\Container\Container', $this);
}

/**
 * Register all of the base service providers.
 * 注册所有的基础服务提供商
 *
 * @return void
 */
protected function registerBaseServiceProviders()
{
    $this->register(new EventServiceProvider($this));

    $this->register(new RoutingServiceProvider($this));
}

/**
 * Set the base path for the application.
 *
 * @param  string  $basePath
 * @return $this
 */
public function setBasePath($basePath)
{
    $this->basePath = $basePath;

    $this->bindPathsInContainer();

    return $this;
}

/**
 * Resolve the given type from the container.
 *
 * (Overriding Container::make)
 *
 * @param  string  $abstract
 * @param  array   $parameters
 * @return mixed
 */
public function make($abstract, $parameters = array())
{
    $abstract = $this->getAlias($abstract);

    if (isset($this->deferredServices[$abstract]))
    {
        $this->loadDeferredProvider($abstract);
    }

    return parent::make($abstract, $parameters);
}
