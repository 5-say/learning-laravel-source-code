PHP 原生函数

/**
 * 返回规范化的绝对路径名。
 * 在 Windows 上，realpath() 会将 unix 风格的路径改成 Windows 风格的。
 * realpath() 扩展所有的符号连接并且处理输入的 path 中的 '/./', '/../' 以及多余的 '/'
 * 并返回规范化后的绝对路径名。
 * 返回的路径中没有符号连接，'/./' 或 '/../' 成分。 
 * realpath() 失败时返回 FALSE，比如说文件不存在的话。
 * 
 * @var string
 */
realpath ( string $path )

/**
 * 把数组中的值赋给一些变量。
 * 像 array() 一样，这不是真正的函数，而是语言结构。list() 用一步操作给一组变量进行赋值。
 * 返回指定的数组。
 * 
 * @var array
 */
list ( mixed $varname [, mixed $... ] )

/**
 * 返回关联数组中当前单元的键名。
 * 返回数组中内部指针指向的当前单元的键名。它不会移动指针。
 * 如果内部指针超过了元素列表尾部，或者数组是空的，key() 会返回 NULL。
 * 
 * @var mixed
 */
key ( array &$array )

/**
 * 返回关联数组中当前单元的值。
 * 返回数组中内部指针指向的当前单元的值。它不会移动指针。
 * 如果内部指针超过了元素列表尾部，或者数组是空的，current() 会返回 FALSE。
 * 
 * @var mixed
 */
current ( array &$array )
